Data Preprocessing

  • Cleaning and preprocessing steps for text data.
  • Tokenization, removal of HTML tags, lowercase conversion, and punctuation removal.
  • Filtering out stop words to focus on meaningful content.
clean_text <- function(text) {
  # Tokenize text
  tokens <- unlist(strsplit(text, "\\s+"))
  # Remove HTML tags
  cleaned_tokens <- gsub("<.*?>", "", tokens)
  # Convert to lowercase
  cleaned_tokens <- tolower(cleaned_tokens)
  # Remove special characters and punctuation
  cleaned_tokens <- gsub("[^a-zA-Z\\s]", "", cleaned_tokens)
  # Remove stop words
  cleaned_tokens <- cleaned_tokens[!cleaned_tokens %in% stopwords("en")]
  return(cleaned_tokens)
}
